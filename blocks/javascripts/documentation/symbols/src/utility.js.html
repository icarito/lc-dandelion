 15 /**
 16  * @class
 17  * @name Util
 18  * @description Static methods on the jQuery object
 19  */
 20 jQuery.extend({
 21     /**
 22      * @lends Util
 23      */ 
 24 
 25     /**
 26      * @description sends str to console, if there is a console
 27      * @returns null
 28      */ 
 29     print: function(str){
 30         if (console && console.log){
 31             console.log(str);
 32         }
 33     },
 34     /**
 35      * @description  tests to see if obj is in array
 36      * @returns boolean
 37      */ 
 38     contains: function(obj, array){
 39         var i;
 40         var len = array.length;
 41         for (i = 0; i < len; i++){
 42             if (array[i] === obj) return true;
 43         }
 44         return false;
 45     },
 46     /**
 47      * @description tests to see if str consists of only integers
 48      * @return boolean
 49      */ 
 50     isInteger: function(str){
 51         return (str.toString().search(/^-?[0-9]+$/) == 0);
 52     },
 53     /**
 54      * @description adds all numbers in list_of_numbers together
 55      * @return Number
 56      */ 
 57     sum: function(list_of_numbers){
 58         var value = 0;
 59         jQuery.each(list_of_numbers, function(){value += this});
 60         return value;
 61     },
 62     /**
 63      * @description returns largest number in list_of_numbers
 64      * @return Number
 65      */ 
 66     max: function(list_of_numbers){
 67         var value = 0;
 68         jQuery.each(list_of_numbers, function(){ if (this > value) value = this; });
 69         return value;
 70     },
 71     /**
 72      * @description returns all property names of obj
 73      * @return Array
 74      */ 
 75     keys: function(obj){
 76         var k = [];
 77         for (key in obj){
 78             k.push(key);
 79         }
 80         return '[' + k.join(', ') + ']';
 81     },
 82     /**
 83      * @static
 84      * @description returns all property names of obj
 85      * @return Array
 86      */ 
 87     upcap: function(str){
 88         return str[0].toUpperCase() + str.slice(1);
 89     }
 90 });

 92 // Additions to the result set object
 93 
 94 $.fn.extend({
 95     // set or return position
 96     positionRelative: function(pos){
 97         if (pos){
 98             this.css({position: 'absolute', left: pos.left + 'px', top: pos.top + 'px'});
 99             return this;
100         }else{
101             return this.offset();
102         }
103     },
104     repositionInFrame: function(frame){
105         var pos = frame.offset();
106         this.each(function(){
107             this.style.position = 'absolute';
108             if (this.parentNode.id != frame.id){
109                 this.style.left = (parseInt(this.style.left) - pos.left) + 'px';
110                 this.style.top = (parseInt(this.style.top) - pos.top) + 'px';
111             }
112         });
113         return this;
114     },
115     // positioning helper, returns {left, top, right, bottom}
116     box: function(){
117         var pos = this.offset();
118         if (!pos){
119             $.print('warning: trying to get offset failed (is object not visible?)');
120             $.print(this);
121             pos = {left: 0, top: 0};
122         }
123 //        console.log(this.info() + '.box() called.  offset == ' + pos);
124         return {left: pos.left, 
125                 top: pos.top, 
126                 right: pos.left + this.width(), 
127                 bottom: pos.top + this.height(),
128                 intersects:function(box2){
129                     if (this.left > box2.right) return false;
130                     if (this.right < box2.left) return false;
131                     if (this.top > box2.bottom) return false;
132                     if (this.bottom < box2.top) return false;
133                     return true;
134                 },
135                 toString: function(){
136                     return 'box{left: ' + this.left + ', top: ' + this.top + ', right: ' + this.right + ', bottom: ' + this.bottom + '}';
137                 }
138             };
139                 
140     },
141     // positioning helper, returns boolean
142     contains: function(other){
143         var b1 = this.box();
144         var b2 = $(other).box();
145         if (b2.left < b1.left) return false;
146         if (b2.top < b1.top) return false;
147         if (b2.right > b1.right) return false;
148         if (b2.bottom > b1.bottom) return false;
149         return true;
150     },
151     // positioning helper, returns boolean
152     intersects: function(other){
153         if (other.length < 1){
154             return false;
155         }
156         return this.box().intersects(other.box());
157     },
158     // go up the ancestor tree until an element matching expr is found
159     up: function(expr){
160         var elem = this;
161 //        console.log('up from ' + this.info() + ' to ' + expr);
162         while(true){
163             elem = elem.parent();
164 //            console.log('up testing ' + elem.info() + ' against ' + expr);
165             if (elem.is(expr)){
166                 return elem;
167             }else if(elem.is('body')){
168                 return $();
169             }
170         }
171     },
172     // debug helper
173     log: function(str){
174         console.log(str);
175         return this;
176     },
177     // utility returning jQuery object containing only first matched element
178     first: function(){
179         return this.eq(0);
180     },
181     // debug helper
182     info: function(){
183         var self = this.get(0);
184         if (!self){
185             console.log('trying to get info on an empty list?');
186             return 'nil object';
187         }
188         return self.nodeName + '#' + self.id + '.' + self.className.split(' ').join('.') + ' (' + this.label() + ')';
189     },
190     // debug helper
191     infoall: function(){
192         var output = ['matched ' + this.length + ' elements: '];
193         this.each(function(){output.push('\t' + $(this).info())});
194         return output.join('\n');
195     },
196     // debug helper
197     loginfo: function(){
198         this.log(this.info());
199         return this;
200     },
201     // utility, return this node and all siblings that come after it
202     subsequent: function(){
203         var id = this.get(0).id;
204         return $('#' + id + ', #' + id + ' ~ ' + '.block');
205     },
206     // Add methods to elements, OO-style (expando version)
207     methods: function(obj){
208         this.each(function(){
209             $.extend(this, obj);
210         });
211     },
212     print_position: function(str, d){
213         var p = d.position();
214         var o = d.offset();
215         var n = d[0];
216         $.print(str + ' top: ' + p.top + ' (' + o.top + '), left: ' + p.left + ' (' + o.left + '), display: ' + d.css('display') + ', parent: ' + n.parentNode.id + '/' + n.parentNode.className);
217     }
218 });
