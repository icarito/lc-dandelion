 54     /**
 55      * @description parses the spec for a block label and returns the parts, including embedded expression slots
 56      * @returns jQuery
 57      */
 58     parseSpec: function(str){
 59         // Handle label strings with embedded content notation
 60         // [] is used for element content
 61         // [#] where # is some numeral: create an IntExpr hole with a default value of #
 62         // [{flag}] double brackets insert a named graphic
 63         // [true] or [false]: create a BoolExpr hold with a default value
 64         // [int]: create an IntExpr hole with no default value
 65         // [bool]: create a BoolExpr hole with no default value
 66         // [color]: create a color picker control
 67         // [key]: create a keypress drop-down list
 68         // [sprite]: create a list of sprites (besides the current sprite)
 69         // [sound]: create a list of available sounds (default + added)
 70         // [check]: make a checkbox, [checked] make a checkbox that defaults to selected
 71         // ["Hello"] or ["Hmm..."]: make a string control from quoted content (type "string")
 72         // [message] add a list of available messages
 73         // [costume] adds a list available costumes
 74         // [effect] list of graphic effects
 75         // the remaining content of the label is type "text"
 76         var parts = str.split(/\]|\[/g);
 77         var part = null;
 78         for (var i = 0; i< parts.length; i++){
 79             part = parts[i];
 80             if ($.isInteger(part)){
 81                 parts[i] = $('<input type="text" class="int" value="'  + part + '" />');
 82             }else if (part[0] === '{'){
 83                 parts[i] = application.images[part.slice(1,-1)].clone().css('vertical-align','middle');
 84             }else if (part == 'true' || part == 'false'){
 85                 parts[i] = $('<input type="text" class="bool" value="' + part + '" />');
 86             }else if(part == 'bool'){
 87                 parts[i] = $('<input type="text" class="bool" value="" readonly="readonly" />');
 88             }else if(part == 'int'){
 89                 parts[i] = $('<input type="text" class="int" value="" />');
 90             }else if(part == 'color'){
 91                 parts[i] = $.colorPicker();
 92             }else if(part == 'key'){
 93                 parts[i] = $.keyList();
 94             }else if(part == 'sprite'){
 95                 parts[i] = $.makeSelect(['Sprite 1']);
 96             }else if(part == 'sound'){
 97                 parts[i] = $.makeSelect(['pop']);
 98             }else if(part == 'message'){
 99                 parts[i] = $.makeSelect(['Add a new message...']);
100             }else if(part == 'costume'){
101                 parts[i] = $.makeSelect(['Costume 1']);
102             }else if(part == 'effect'){
103                parts[i] = $.makeSelect(['fisheye']);
104             }else if (part[0] == '"'){
105                 parts[i] = $('<input type="text" class="string_input" value="' + part.slice(1,-1) + '" />'); 
106             }else if (part == 'check'){ 
107                 parts[i] = $('<input type="checkbox" />');
108             }else if (part == 'checked'){ 
109                 parts[i] = $('<input type="checkbox" checked="checked" />');
110             }else{ 
111                 if (part){
112                     parts[i] = $('<span>' + part + '</span>');
113                 }else{
114                     parts[i] = null; // no need to wrap empty strings
115                 }
116             }
117         }
118         return parts;
119     }

214 Block.prototype.label = function(str){
215     if (str){
216         var label_elements = $.parseSpec(str);
217         for (var i = 0; i < label_elements.length; i++){
218             if (label_elements[i]){ // don't try to append nulls
219                 try{
220                     this._label.append(label_elements[i]);
221                 }catch(e){
222                     console.log('problem adding element ' + label_elements[i].get(0).nodeName + ' to block label');
223                 }
224             }
225         }
226         return this;
227     }else{
228         return this._label.text();
229     }

251 /**
252  * @param {Block} block
253  * @description Adds a block to this block, as the next in a sequence
254  * @returns {Block}
255  */
256 Block.prototype.append = function(block){
257 //    this.drag_wrapper.css('border', '1px solid red');
258 //    $.print('trying to append ' + block + ' to ' + this);
259     if (this == block) return;
260     block.drag_wrapper.css({left: 0, top: 0, position: 'relative'});
261     if (this.next){
262         block.append(this.next);
263     }
264     this.drag_wrapper.append(block.drag_wrapper);
265     block.drag_parent = this;
266     this.next = block;
267     return this;

275     var canvas = $('#scripts_container');
276     var elem = this.drag_wrapper;

278     var coff = canvas.offset();
279     canvas.append(elem);

281     return this;

284 /**
285  * @param {Block} other
286  * @description Tests to see if the either blocks' drop pointer interesects the other's drop target. Highlights the drop target if so, and returns true, otherwise returns false.

288  */
289 Block.prototype.test_snapping = function(other){
290     if (this.drop_intersects(other)){
291         this.highlight_drop_pointer(true, other);
292         return true;
293     }else if(other.drop_intersects(this)){
294         other.highlight_drop_pointer(true, this);
295         return true;
296     }else{
297         other.highlight_drop_target(false, other);
298         return false;
299     }

314 /**
315  * @description during a drag, test to see if this block should snap to any other block.
316  * @returns {null}
317  */
318 Block.prototype.ondrag = function(){
319     var self = this;
320     if (!Block.blocks.length) return;
321     var matched = false
322     $.each(Block.blocks, function(idx, block){
323         matched = self.test_snapping(block);
324         if (matched){
325             self.snap_target = block;
326             return false // stop the iteration
327         }
328     });
329     if (!matched){
330         self.highlight_drop_target(false, self);
331     }
332 };

334 /**
335  * @description At the end of a drag, if we are outside script canvas, remove block. Otherwise, if block is a factory, put new block into position in script canvas. If block is passes the snapping test, snap it with other block.  If already snapped to another block, but it fails snapping test, unsnap.
336  * @returns {null}
337  */
338 Block.prototype.ondragend = function(){
339     var script_canvas = $('#scripts_container');
340     // Check to see if we're in the script canvas at all
341     if (!(this.snap_target || script_canvas.intersects(this.get_helper().intersection_shape()))){
342         return;
343     }
344     this.highlight_drop_pointer(false, this);
345     if (this.snap_target){
346         this.snap_target.highlight_drop_pointer(false, this.snap_target);
347     }
348     var instance = this;
349     if (!this.isInstance){
350         instance = this.clone();
351     }
352     if (!this.isInstance){ // reposition if factory
353         script_canvas.append(instance.drag_wrapper);
354         instance.drag_wrapper.repositionInFrame(script_canvas);
355     }else{
356         if(this.drag_parent){
357             instance.moveToScriptCanvas();
358         }
359     }
360     if (this.snap_target){
361         if (instance.drop_intersects(this.snap_target)){
362             this.snap_target.append(instance);
363         }else if(this.snap_target.drop_intersects(instance)){
364             instance.append(this.snap_target);
365         }
366     }
367     if (!this.isInstance && this.current_helper){
368         this.current_helper.hide();
369     }
370     $.ui.ddmanager.current.cancelHelperRemoval = true;
371 }

375 /**
376  * @description Returns the drag_wrapper element, wrapped in jQuery.  If block is a factory, create a new drag_helper to drag.
377  * @returns {jQuery} drag_wrapper
378  */
379 Block.prototype.get_helper = function(){
380     if (this.isInstance){
381         return this.drag_wrapper;
382     }else{
383         if (!this.current_helper){
384             this.current_helper = this.drag_wrapper.clone(true);
385             this.drop_target = this.current_helper.find('.drop_target');
386             this.drop_pointer = this.current_helper.find('.drop_pointer');
387         }else{
388             this.current_helper.css('display', 'block');
389         }
390         $(document.body).append(this.current_helper);
391         return this.current_helper;
392     }
393 };

564 /**
565  * @param {Block} other
566  * @description Tests to see if one object intersects (drop_pointer to drop_target) another
567  * @returns {Boolean}
568  */
569 Block.prototype.drop_intersects = function(other){
570     if (other === this) return false;
571     if (!this.drop_pointer) return false;
572     if (!other.drop_target) return false;
573     if (this.drag_parent == other) return false;
574     if (other.drag_parent == this) return false;
575     return this.drop_pointer.intersects(other.drop_target);
576 }

771 /**
772  * @class
773  * @name Loop
774  * @extends Block
775  */
776 function Loop(params){
777     this.initialize(params);
778     this.next_in_loop = null;
779     this.block.addClass('loop container containable');
780     this.block.prepend('<div class="top_left"></div>' + 
781         '<div class="top_right"></div>' + 
782         '<div class="left"></div>' + 
783         '<div class="bottom_left"></div>' + 
784         '<div class="bottom"></div>' + 
785         '<div class="bottom_right"></div>'
786     );    
787     this.handle = $('<div class="top"></div>');
788     this.block.prepend(this.handle);
789     this.make_containable();
790 }
791 Loop.prototype = new Block();
792 Loop.prototype.constructor = Loop;
793 
794 Loop.prototype.appendLoop = function(block){
795     if (this.next_in_loop){
796         try{
797             this.next_in_loop.append(block);
798         }catch(e){
799             console.log('Here I am, trying to append a simple block: ' + e.message);
800             console.log('next in loop: ' + this.next_in_loop.toString());
801             console.log('appending block: ' + block.toString());
802         }
803     }else{
804         this.block.append(block.drag_wrapper);
805         this.next_in_loop = block;
806         //console.log('setting next in loop: ' + block.toString());
807     }
808     return this;
809 }
