 27 /**
 28  * @class
 29  * @name Layout
 30  */
 31 $.fn.extend( /** @lends Layout.prototype */ {
 32     /**
 33      * @returns Element
 34      * @description creates a new div, gives it an id, places it absolutely, appends it to current item, and returns the new div (NOT current item)
 35      */
 36     place: function(box){
 37         var div = $('<div id="' + box.name + '"></div>');
 38         div.css({position: 'absolute', left: box.x, top: box.y, width: box.w, height: box.h, backgroundColor: box.c});
 39         this.append(div);
 40         return div;
 41     },
 42     /**
 43      * @returns jQuery
 44      * @description applies CSS border rounding, border, and overflow, returns this
 45      */
 46     round: function(topleft, topright, bottomright, bottomleft, bordercolor){
 47         this.css('-webkit-border-top-left-radius', topleft);
 48         this.css('-moz-border-radius-topleft', topleft);
 49         this.css('border-top-left-radius', topleft);
 50         this.css('-webkit-border-top-right-radius', topright);
 51         this.css('-moz-border-radius-topright', topright);
 52         this.css('border-top-right-radius', topright);
 53         this.css('-webkit-border-bottom-right-radius', bottomright);
 54         this.css('-moz-border-radius-bottomright', bottomright);
 55         this.css('border-bottom-right-radius', bottomright);
 56         this.css('-webkit-border-bottom-left-radius', bottomleft);
 57         this.css('-moz-border-radius-bottomleft', bottomleft);
 58         this.css('-border-bottom-left-radius', bottomleft);
 59         this.css('border', '3px inset ' + bordercolor);
 60         this.css('overflow-x', 'auto');
 61         return this;
 62     },
 63     /**
 64      * @returns jQuery
 65      * @description uses button_spec to create a button menu interface, returns this
 66      */
 67     block_menu_buttons: function(button_spec){
 68         var parent = this;
 69         $.each(button_spec, function(idx, spec){
 70             var button = $('<div id="button_' + this.name + '">' + $.upcap(this.name) + '</div>');
 71             parent.append(button);
 72             button.round(10,10,10,10,md_grey);
 73             button.css({backgroundColor: dk_grey, color: spec.color, width: 75, display: 'block', textAlign: 'center', 'float': 'left', marginLeft: 10, borderStyle: 'outset'});
 74             button.click(function(){
 75                 application.show_block_palette(spec.name);
 76                 $(application.current_button).trigger('unselect');
 77                 application.current_button = this;
 78                 $(this).css({backgroundColor: spec.color, color: white});
 79             });
 80             button.bind('unselect', function(){
 81                 $(this).css({backgroundColor: dk_grey, color: spec.color});
 82             });
 83             if (spec['default']){
 84                 button.click();
 85             }
 86         });
 87         return this;
 88     },
 89     /**
 90      * @returns jQuery
 91      * @description creates a new block palette and hides it (they all occupy the same space), return this
 92      */
 93     add_palette: function(name, color, block_spec){
 94         var palette = $('<div></div>');
 95         application['_' + name + '_palette'] = palette;
 96         this.append(palette);
 97         $.each(block_spec, function(){
 98             palette.append( new this.type({label: this.label, x: 2, y: this.y, color: color}).drag_wrapper);
 99         });
100         palette.hide();
101         return this;
102     }
103     
104 });
